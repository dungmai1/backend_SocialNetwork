# Mức log mặc định: chỉ in từ INFO trở lên
logging.level.root=INFO

# Spring: giữ INFO thôi (không in DEBUG)
logging.level.org.springframework=INFO
logging.level.org.springframework.web=INFO
logging.level.org.springframework.security=INFO

# Log cho package của bạn (nếu chỉ cần cơ bản)
logging.level.SocialNetwork.SocialNetwork=INFO

# HTTP request/response: chỉ WARN để hạn chế log
logging.level.org.springframework.web.filter.CommonsRequestLoggingFilter=WARN

# Hibernate: chỉ INFO (không in SQL và bind param chi tiết)
logging.level.org.hibernate.SQL=INFO
logging.level.org.hibernate.type.descriptor.sql.BasicBinder=INFO

# Ghi log ra file
logging.file.name=logs/app.log
